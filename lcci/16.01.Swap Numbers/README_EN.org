#+begin_src python
  class Solution:
      def swapNumbers(self, numbers: List[int]) -> List[int]:
          numbers[0], numbers[1] = numbers[1], numbers[0]
          return numbers
#+end_src

*** *Java*
    :PROPERTIES:
    :CUSTOM_ID: java
    :END:
#+begin_src java
  class Solution {
      public int[] swapNumbers(int[] numbers) {
          numbers[0] = numbers[0] ^ numbers[1];
          numbers[1] = numbers[0] ^ numbers[1];
          numbers[0] = numbers[0] ^ numbers[1];
          return numbers;
      }
  }
#+end_src
