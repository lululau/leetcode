#+begin_src python
  class Solution:
      def isUnique(self, astr: str) -> bool:
          bitmap = 0
          for c in astr:
              pos = ord(c) - ord('a')
              if (bitmap & (1 << pos)) != 0:
                  return False
              bitmap |= (1 << pos)
          return True
#+end_src

  class Solution {
      public boolean isUnique(String astr) {
          int bitmap = 0;
          for (char c : astr.toCharArray()) {
              int pos = c - 'a';
              if ((bitmap & (1 << pos)) != 0) {
                  return false;
              }
              bitmap |= (1 << pos);
          }
          return true;
      }
  }
