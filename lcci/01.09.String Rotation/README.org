#+begin_src python
  class Solution:
      def isFlipedString(self, s1: str, s2: str) -> bool:
          return len(s1) == len(s2) and s1 in (s2 * 2)
#+end_src

#+begin_src java
  class Solution {
      public boolean isFlipedString(String s1, String s2) {
          return s1.length() == s2.length() && (s2 + s2).indexOf(s1) != -1;
      }
  }
