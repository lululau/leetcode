#+begin_src python
  class Solution:
      def waysToStep(self, n: int) -> int:
          if n < 3:
              return n
          a, b, c = 1, 2, 4
          for _ in range(4, n + 1):
              a, b, c = b, c, (a + b + c) % 1000000007
          return c
#+end_src

#+begin_src java
  class Solution {
      public int waysToStep(int n) {
          if (n < 3) {
              return n;
          }
          int a = 1, b = 2, c = 4;
          for (int i = 4; i <= n; ++i) {
              int t = a;
              a = b;
              b = c;
              c = ((a + b) % 1000000007 + t) % 1000000007;
          }
          return c;
      }
  }
#+end_src

  class Solution {
  public:
      int waysToStep(int n) {
          if (n < 3) {
              return n;
          }
          int a = 1, b = 2, c = 4, i = 4;
          while (i++ <= n) {
              int t = ((a + b) % 1000000007 + c) % 1000000007;
              a = b;
              b = c;
              c = t;
          }
          return c;
      }
  };
