异或求解。两个相同的数异或的结果为 0。

也可以用数学求解。求出 =[0..n]=
的和，减去数组中所有数的和，就得到了缺失的数字。

#+begin_src python
  class Solution:
      def missingNumber(self, nums: List[int]) -> int:
          res = len(nums)
          for i, v in enumerate(nums):
              res ^= (i ^ v)
          return res
#+end_src

#+begin_src java
  class Solution {
      public int missingNumber(int[] nums) {
          int res = nums.length;
          for (int i = 0, n = res; i < n; ++i) {
              res ^= (i ^ nums[i]);
          }
          return res;
      }
  }
#+end_src

- 数学

#+begin_src java
  class Solution {
      public int missingNumber(int[] nums) {
          int res = nums.length;
          for (int i = 0, n = res; i < n; ++i) {
              res += (i - nums[i]);
          }
          return res;
      }
  }
#+end_src
