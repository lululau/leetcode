*** *SQL*
    :PROPERTIES:
    :CUSTOM_ID: sql
    :END:
#+begin_src sql
  # Write your MySQL query statement below
  SELECT
      user_id, count(1) AS followers_count
  FROM
      Followers
  GROUP BY
      user_id
  ORDER BY
      user_id
#+end_src
