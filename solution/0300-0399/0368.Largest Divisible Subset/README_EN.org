subset answer such that every pair (answer[i], answer[j]) of elements in
this subset satisfies:

#+begin_html
  </p>
#+end_html

#+begin_html
  <ul>
#+end_html

#+begin_html
  <li>
#+end_html

answer[i] % answer[j] == 0, or

#+begin_html
  </li>
#+end_html

#+begin_html
  <li>
#+end_html

answer[j] % answer[i] == 0

#+begin_html
  </li>
#+end_html

#+begin_html
  </ul>
#+end_html

#+begin_html

#+end_html
