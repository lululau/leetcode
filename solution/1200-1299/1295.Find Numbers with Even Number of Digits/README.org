#+begin_src python
  class Solution:
      def findNumbers(self, nums: List[int]) -> int:
          res = 0
          for num in nums:
              res += (len(str(num)) & 1) == 0
          return res
#+end_src

#+begin_src java
  class Solution {
      public int findNumbers(int[] nums) {
          int res = 0;
          for (int num : nums) {
              res += (String.valueOf(num).length() & 1) == 0 ? 1 : 0;
          }
          return res;
      }
  }
