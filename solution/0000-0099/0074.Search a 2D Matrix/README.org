#+begin_src python
  class Solution:
      def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
          m, n = len(matrix), len(matrix[0])
          l, h = 0, m * n - 1
          while l <= h:
              mid = (l + h) >> 1
              x, y = divmod(mid, n)
              if matrix[x][y] == target:
                  return True
              if matrix[x][y] < target:
                  l = mid + 1
              else:
                  h = mid - 1
          return False
#+end_src

          int l = 0, h = m * n - 1;
          while (l <= h) {
              int mid = (l + h) >>> 1;
              int x = mid / n, y = mid % n;
              if (matrix[x][y] == target) return true;
              if (matrix[x][y] < target) l = mid + 1;
              else h = mid - 1;
          }
          return false;
      }
  }

      if (matrix[x][y] < target) {

      }
