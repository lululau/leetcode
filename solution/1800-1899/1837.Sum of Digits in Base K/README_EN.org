#+begin_src python
  class Solution:
      def sumBase(self, n: int, k: int) -> int:
          res = 0
          while n != 0:
              n, t = divmod(n, k)
              res += t
          return res
#+end_src

#+begin_src java
  class Solution {
      public int sumBase(int n, int k) {
          int res = 0;
          while (n != 0) {
              res += (n % k);
              n /= k;
          }
          return res;
      }
  }
#+end_src
