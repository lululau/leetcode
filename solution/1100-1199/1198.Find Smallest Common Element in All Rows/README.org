  class Solution:
      def smallestCommonElement(self, mat: List[List[int]]) -> int:
          counter = collections.Counter()
          for row in mat:
              for num in row:
                  counter[num] += 1
                  if counter[num] == len(mat):
                      return num
          return -1
#+end_src

  class Solution {
      public int smallestCommonElement(int[][] mat) {
          int[] counter = new int[10001];
          for (int[] row : mat) {
              for (int num : row) {
                  ++counter[num];
                  if (counter[num] == mat.length) {
                      return num;
                  }
              }
          }
          return -1;
      }
  }
