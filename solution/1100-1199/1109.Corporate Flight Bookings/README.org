#+begin_src python
  class Solution:
      def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
          delta = [0] * n
          for first, last, seats in bookings:
              delta[first - 1] += seats
              if last < n:
                  delta[last] -= seats
          for i in range(n - 1):
              delta[i + 1] += delta[i]
          return delta
#+end_src

  class Solution {
      public int[] corpFlightBookings(int[][] bookings, int n) {
          int[] delta = new int[n];
          for (int[] booking : bookings) {
              int first = booking[0], last = booking[1], seats = booking[2];
              delta[first - 1] += seats;
              if (last < n) {
                  delta[last] -= seats;
              }
          }
          for (int i = 0; i < n - 1; ++i) {
              delta[i + 1] += delta[i];
          }
          return delta;
      }
  }

      let delta = new Array(n).fill(0);
      for (let book of bookings) {
          let [start, end, num] = book;
          start -= 1;
          delta[start] += num;
          if (end != n) {
              delta[end] -= num;
          }
      }
      for (let i = 1; i < n; i++) {
          delta[i] += delta[i - 1];
      }
      return delta;
  };
