*** *SQL*
    :PROPERTIES:
    :CUSTOM_ID: sql
    :END:
#+begin_src sql
  # Write your MySQL query statement below
  SELECT
      DISTINCT(author_id) as id
  FROM
      Views
  WHERE
      author_id = viewer_id
  ORDER BY
      id;
#+end_src
