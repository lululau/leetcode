#+begin_src python
  class Solution:
      def isArmstrong(self, n: int) -> bool:
          k = len(str(n))
          s, t = 0, n
          while t:
              t, v = divmod(t, 10)
              s += pow(v, k)
          return n == s
#+end_src

#+begin_src java
  class Solution {
      public boolean isArmstrong(int n) {
          int k = String.valueOf(n).length();
          int s = 0, t = n;
          while (t != 0) {
              s += Math.pow(t % 10, k);
              t /= 10;
          }
          return n == s;
      }
  }
#+end_src

  /**
   * @param {number} n

  var isArmstrong = function (n) {
    const k = String(n).length;
    let s = 0;
    let t = n;
    while (t) {
      s += Math.pow(t % 10, k);
      t = Math.floor(t / 10);
    }
    return n == s;
  };
