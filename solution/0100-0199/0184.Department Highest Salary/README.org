*** *SQL*
    :PROPERTIES:
    :CUSTOM_ID: sql
    :END:
#+begin_example
  SELECT
      Department.NAME AS Department,
      Employee.NAME AS Employee,
      Salary
  FROM
      Employee,
      Department
  WHERE
      Employee.DepartmentId = Department.Id
      AND ( Employee.DepartmentId, Salary )
      IN (SELECT DepartmentId, max( Salary )
          FROM Employee
          GROUP BY DepartmentId )
#+end_example
