*** *Python3*

    :END:
#+begin_src python
  class Solution:
      def titleToNumber(self, columnTitle: str) -> int:
          res = 0
          for c in columnTitle:
              res = res * 26 + (ord(c) - ord('A') + 1)
          return res
#+end_src

*** *Java*

#+begin_src java
  class Solution {
      public int titleToNumber(String columnTitle) {
          int res = 0;
          for (char c : columnTitle.toCharArray()) {
              res = res * 26 + (c - 'A' + 1);
          }
          return res;
      }
  }
#+end_src
