利用异或运算的规律找出不同的位

- 0 ^ 0 = 0
- 1 ^ 1 = 0
- 0 ^ 1 = 1
- 1 ^ 0 = 1

#+begin_src python
  class Solution:
      def hammingDistance(self, x: int, y: int) -> int:
          num, count = x ^ y, 0
          while num != 0:
              num &= num - 1
              count += 1
          return count
#+end_src

#+begin_src java
  class Solution {
      public int hammingDistance(int x, int y) {
          int num = x ^ y;
          int count = 0;
          while (num != 0) {
              num &= num - 1;
              count++;
          }
          return count;
      }
  }
#+end_src

或者利用库函数 =Integer.bitCount()=

#+begin_src java
  class Solution {
      public int hammingDistance(int x, int y) {
          return Integer.bitCount(x ^ y);
      }
  }
#+end_src

  /**
   * @param {number} x
   * @param {number} y
   * @return {number}
   */
  var hammingDistance = function(x, y) {
      let distance = x ^ y;
      let count = 0;
      while (distance != 0) {
          count++;
          distance &= (distance - 1);
      }
      return count;
  };
