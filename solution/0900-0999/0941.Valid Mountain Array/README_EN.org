mountain array.

#+begin_html
  </li>
#+end_html

#+begin_html
  <li>
#+end_html

#+end_html

#+end_html

#+begin_html
  </li>
#+end_html

#+begin_html
  <li>
#+end_html

arr[i] > arr[i + 1] > ... > arr[arr.length - 1]
